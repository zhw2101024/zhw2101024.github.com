静态网页显示当前时间

页面头部的时间是用JavaScript实现的，显示当前时间，精确到秒并显示星期、时区。其实很简单，就是利用JavaScript内置的Date对象，然后用setInterval实现没秒钟刷新一次时间。 代码如下：

<span class="timeprompt">当前时间：</span><br />
<span id="currentDate"></span><br />
<span id="currentTime"></span><br />
<script type="text/javascript">
  var weekDays = ['日','一','二','三','四','五','六'];
  var current = new Date();
  var year = current.getFullYear();
  var month = current.getMonth();
  var date = current.getDate();
  var day = current.getDay();
  var datestr = year + '年' + (month + 1) + '月' + date + '日 星期' + weekDays[day];
  var dateobj = document.getElementById("currentDate");
  dateobj.innerHTML = datestr;
  var hours,minutes,seconds,timestr;
  var timeobj = document.getElementById("currentTime");
  var tz = current.getTimezoneOffset() / -60;
  function flushtime() {
    var current = new Date();
    hours = current.getHours();
    minutes = current.getMinutes();
    seconds = current.getSeconds();
    timestr = hours + '时' + minutes + '分' + seconds + '秒 东' + tz + '区';
    timeobj.innerHTML = timestr;
  }
  flushtime();
  timeobj.innerHTML = timestr;
  setInterval(function() {flushtime();},1000);
</script>
